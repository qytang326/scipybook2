from scipy.sparse import coo_matrix
inner_count = len(x0)
edge_count = len(x_edge)

r = np.r_[x0, x0, x0, x0, x0, x_edge]   
c = np.r_[x0, x1, x2, x3, x4, x_edge]   
v = np.ones(inner_count*5 + edge_count) 
v[:inner_count] = -4                    

A = coo_matrix((v, (r, c))).tocsc()

result = dst.copy()

for ch in (0, 1, 2): #
